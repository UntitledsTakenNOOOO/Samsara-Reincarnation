actor "RMR Grenade Launcher" : Weapon
{
	Weapon.SelectionOrder 200
	Weapon.SlotNumber 5
	Weapon.BobStyle Inverse
	Weapon.BobRangeX 0.85
	Weapon.BobSpeed 1.1	
	Weapon.AmmoType "RocketAmmo"
	Weapon.AmmoGive 9
	Weapon.AmmoUse 1
	Inventory.PickupSound "RMR/pickup" 		
	-WEAPON.NOAUTOFIRE
	+WEAPON.EXPLOSIVE
	Tag "Grenade Launcher"	
	Inventory.PickupMessage "You picked up a Grenade Launcher!"
	Inventory.RestrictedTo "RebelMoonCommando"
	Tag "Sukayev T-23 Grenade Launcher"
	States
	{
	  Ready:
		RM9Y A 1 A_WeaponReady
		Loop

	  Deselect:
		TNT1 A 0 A_Lower
		RM9Y A 1 A_Lower
		Loop

	  Select:
		TNT1 A 0 A_Raise
		RM9Y A 1 A_Raise
		Loop

	  Fire:
		TNT1 A 0 A_PlaySound("RMR/PIGFIRE")
		TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"CoopFire")
		RM9Y B 1 bright A_FireCustomMissile("RMRDISCOBALL")
		RM9Y CDDD 3 Bright
		RM9Y CB 4 Bright
		RM9Y AA 5 Bright
		Goto Ready
		
	  CoopFire:
		RM9Y B 1 bright A_FireCustomMissile("COOPRMRDISCOBALL")
		RM9Y CDDD 3 Bright
		RM9Y DD 3 Bright
		RM9Y CB 4 Bright
		RM9Y AA 5 Bright
		Goto Ready
	  Spawn:
		RM9Y E -1 Bright
		Loop
	}
}

	DeathSound ""
	Scale 0.40
	Radius 5

	+THRUGHOST
	+FORCEXYBILLBOARD
	+FORCERADIUSDMG

	DamageType "RMR-DiscoBall"
	States
	{
	  Spawn:
		RMZ2 NO 1 bright
		Loop

	  Death:
		TNT1 A 0 A_Explode(2000,64,!XF_HURTSOURCE)

		TNT1 A 5 A_PlaySound("RMR/DiscoBoom")
		Stop
	}
}
